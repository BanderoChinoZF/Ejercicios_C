//PSEUDOCODIGO PARA CALCULAR LA SUMA DE LOS PRIMEROS 150 NUMEROS

ALGORITMO "SUMA DE LOS PRIMEROS 150 NUMEROS POSITIVOS"
VAR
	INT numero = 150;
	INT suma;
INICIO
	ESRIBIR("SUMA DE LOS PRIMERO 150 NUMEROS POSITIVOS");

	SI( numero>0)
		
		suma = ((numero*(numero+1))/2);
		ESCRIBIR("LA SUMA DE LOS PRIMEROS 150 NUMEROS ES: " + suma);
	ENTONCES
		ESCRIBIR("EL NUMERO NO ES POSITIVO");
