//PSEUDOCODIGO PARA CALCULAR UN AÑO BISIESTO

ALGORITMO "CALCULAR SI UN AÑO ES BISIESTO O NO"
VAR
	INT anio

INICIO PROGRAMA
	
	ESCRIBIR("PROGRAMA PARA CALCULAR SI UN AÑO ES BISIESTO O NO");

	ESCRIBIR("INTRODUZCA UNA AÑO -> ");
	LEER(anio);

	//UTILIZAMOS EL % PARA OBTENER EL MODULO
	SI(anio%4==0 AND anio%100 != 0 OR anio%400 == 0)
		ESCRIBIR("EL AÑO INGRESADO ES BISIESTO");
	ENTONCES
		ESCRIBIR("EL AÑO NO ES BISIESTO");
		
FIN PROGRAMA
