//PSEUDOCODIGO DEL JUEGO DE PIEDRA, PAPEL Y TIJERAS

ALGORITMO "JUEGO DE ROCA, PAPEL Y TIJERAS"
VAR
	CHAR jug1;
	CHAR jug2;
INICIO
	ESCRIBIR("BIENVENIDO AL JUEGO DE ROCA, PAPEL Y TIJERAS");
	ESCRIBIR("PUEDES ELEGIR DE LA SIGUIENTE MANERA");
	ESCRIBIR(" TIJERAS: 'T' - PAPEL: 'P' - ROCA: 'R' ");

	ESCRIBIR("¿Cual es su eleccion jugador 1?");
	LEER(jug1);

	ESCRIBIR("¿Cual es su leeccion jugador 2?");
	LEER(jug2);

	//DECLARAMOS NUESTRAS ESTRUCTURAS SWITCH-CASE PARA LA LOGICA DEL JUEGO
	SWITCH(jug1)

		CASE 'T':
			SWITCH(jug2)

				CASE 'T':
					ESCRIBIR("SE EMPATO EL JUEGO");
				CASE 'P':
					ESCRIBIR("T-P TIJERA CORTA PAPEL, GANA EL JUGADOR 1");
				CASE 'R':
					ESCRIBIR("T-R ROCA ROMPE TIJERAS, GANA EL JUGADOR 2");
		CASE 'R':
			SWITCH(jug2)

				CASE 'R':
					ESCRIBIR("SE EMPATO EL JUEGO");
				CASE 'P':
					ESCRIBIR("P-R PAPEL ENVUELVE A LA ROCA, GANA JUGADOR 2");
				CASE 'T':
					ESCRIBIR("T-R ROCA ROMPE TIJERAS, GANA JUGADOR 1");
		CASE 'P':
			SWITCH(jug2)

				CASE 'P':
					ESCRIBIR("SE EMPATO EL JUEGO");
				CASE 'R':
					ESCRIBIR("P-R PAPEL ENVUELVE A LA ROCA, GANA EL JUGADOR 1");
				CASE 'T':
					ESCRIBIR("T-P TIJERAS CORTA EL PAPEL, GANA EL JUGADOR 2");
		DEFAULT:
			ESCRIBIR("SELECCIONA UNA OPCION VALIDA");
FIN
